%matplotlib notebook

from san import Batch
from_date = '2018-06-01'
to_date = '2018-10-16'

batch = Batch()
batch.get(
    "daily_active_addresses/santiment",
    from_date=from_date,
    to_date=to_date,
    interval="1d"
)

batch.get(
    "prices/santiment",
    from_date=from_date,
    to_date=to_date,
    interval="1d"
)

[daa, prices] = batch.execute()
print(daa.head(5))
